from pyscript import Element # Constants for passing and Dean's List criteria PASSING_GRADE = 75 DEANS_LIST_CRITERIA = 90 def calculate_grades(event): # Prevent form from submitting event.preventDefault() # Get the Prelim Exam grade prelim_exam = float(Element("prelim_exam").element.value) # Calculate the Prelim grade (since it's 20% of the total) prelim_grade = prelim_exam # Calculate how much grade is needed from Midterm (30%) and Final (50%) combined remaining_grade_needed = PASSING_GRADE - (prelim_grade * 0.2) required_midterm_final = remaining_grade_needed / (0.3 + 0.5) # Calculate required midterm and final grades for passing required_midterm = required_midterm_final * 0.3 required_final = required_midterm_final * 0.5 # Dean's Lister calculation remaining_grade_needed_deans = DEANS_LIST_CRITERIA - (prelim_grade * 0.2) required_midterm_final_deans = remaining_grade_needed_deans / (0.3 + 0.5) required_midterm_deans = required_midterm_final_deans * 0.3 required_final_deans = required_midterm_final_deans * 0.5 # Check if it's possible to pass if required_midterm_final <= 100: pass_message = "You have a chance to pass!" else: pass_message = "It is difficult to pass." # Check if it's possible to be a Dean's Lister if required_midterm_final_deans <= 100: deans_list_message = f"To be a Dean’s Lister, you need at least {required_midterm_deans:.2f}% in the Midterm and {required_final_deans:.2f}% in the Finals." else: deans_list_message = "It is difficult to achieve the Dean’s Lister status." # Update the result in the UI Element("prelimGrade").element.innerHTML = f"{prelim_grade:.2f}" Element("requiredMidterm").element.innerHTML = f"{required_midterm:.2f}" Element("requiredFinal").element.innerHTML = f"{required_final:.2f}" Element("passMessage").element.innerHTML = pass_message Element("deansListMessage").element.innerHTML = deans_list_message Element("result").element.style.display = "block" # Bind the calculate_grades function to the form submission event Element("gradeForm").element.onsubmit = calculate_grades
